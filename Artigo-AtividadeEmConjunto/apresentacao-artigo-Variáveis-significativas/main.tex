\section{A origem.}
\begin{frame}{A origem}{do Índice de Desenvolvimento Humano (IDH).}

\begin{itemize}
			\item 1990, Programa das Nações Unidas (PNUD).
			\item Relatório de Desenvolvimento Humano (RDH).
			\item Prêmio Nobel de Economia de 1998.
            \item Medida global da qualidade de vida e bem estar social.
		\end{itemize}
\begin{figure}[H]
\centering
\includegraphics[scale=0.25]{Images/Mahbub-ul-Haq.jpg} \quad

\label{Mahbub ul Haq e Amartya Sen, 1990}
\end{figure}

\emph{"Desviar o foco do desenvolvimento da economia e da contabilidade de renda nacional para políticas centradas em pessoas."\cite{Haq1995}}

\end{frame}

\section{Cálculo do IDH.}
\begin{frame}{Cálculo do IDH.}

\begin{enumerate}
			\item EV - Expectativa de Vida.
			\item IE - Índice de Educação.
			\item IR - Índice de Renda (Paridade do Poder de Compra).
		\end{enumerate}
\vspace{15pt}        
\begin{center}
            \begin{minipage}{4cm}
                \begin{varblock}[4cm]{}
                    $IDH = \sqrt[3] {EV \cdot IE \cdot IR}$
                \end{varblock}
            \end{minipage}
           \end{center}

\end{frame}
